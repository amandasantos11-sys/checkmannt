function atualizarVisualizacaoLogs() { const totalChecklists = logsSessao.filter(l => l.tipo === "CHECKLIST").length; const totalOS = logsSessao.filter(l => l.tipo === "ORDEM DE SERVIÇO").length; document.getElementById('ind-checklists').innerText = totalChecklists; document.getElementById('ind-os').innerText = totalOS; let totalRespostas = 0; let respostasOk = 0; logsSessao.filter(l => l.tipo === "CHECKLIST").forEach(log => { if(log.detalhes && log.detalhes.toUpperCase().includes("OK")) { respostasOk++; } totalRespostas++; }); if(totalRespostas > 0) { let taxa = Math.round((respostasOk / totalRespostas) * 100); document.getElementById('ind-oee').innerText = taxa + "%"; } const ultimaOS = logsSessao.find(l => l.tipo === "ORDEM DE SERVIÇO"); if(ultimaOS && ultimaOS.detalhes && ultimaOS.detalhes.toUpperCase().includes("ALTA")) { document.getElementById('ind-status').innerText = "BLOQUEADA"; document.getElementById('ind-status').className = "txt-laranja"; } else { document.getElementById('ind-status').innerText = "OPERANDO"; document.getElementById('ind-status').className = "txt-verde"; } // Histórico Geral do Operador document.getElementById('lista-historico-local').innerHTML = logsSessao.map(log => `
${log.data} - ${log.tipo} (Por: ${log.usuario ? log.usuario.toUpperCase() : 'S/U'})

${log.detalhes || ''}

`).join(''); // Feed de Acompanhamento da Manutenção document.getElementById('lista-feed-manutencao').innerHTML = logsSessao.map(log => `
${log.data} - ${log.tipo}
Máquina: ${log.equipamento || ''} | Usuário: ${log.usuario ? log.usuario.toUpperCase() : 'S/U'}

${log.detalhes || ''}

`).join(''); // CORREÇÃO DO BACKLOG DA MANUTENÇÃO (Busca sem distinção de maiúsculas/minúsculas) const apenasOS = logsSessao.filter(l => l.tipo === "ORDEM DE SERVIÇO"); if(apenasOS.length === 0) { document.getElementById('lista-backlog-manutencao').innerHTML = `

Nenhuma O.S. pendente.

`; } else { document.getElementById('lista-backlog-manutencao').innerHTML = apenasOS.map(log => { let criticidadeTag = "tag-baixa"; let critTxt = "baixa"; // Transforma o texto em maiúsculo temporariamente apenas para testar a palavra const detalhesMaiusculo = log.detalhes ? log.detalhes.toUpperCase() : ""; if(detalhesMaiusculo.includes("ALTA") || detalhesMaiusculo.includes("CRÍTICA") || detalhesMaiusculo.includes("CRITICA")) { criticidadeTag = "tag-alta"; critTxt = "alta"; } else if(detalhesMaiusculo.includes("MEDIA") || detalhesMaiusculo.includes("MÉDIA")) { criticidadeTag = "tag-media"; critTxt = "média"; } return `
${critTxt} ${log.data} - ${log.equipamento || 'Máquina Geral'}

${log.detalhes || ''}

`; }).join(''); } }
